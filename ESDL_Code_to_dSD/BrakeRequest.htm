calc (BLS::log;FBLS::log;pMc::cont;pMc_available::log;pTgtVeh::cont;pTgtVeh_Enable::log;BrakeRqst_Parking::log;p_Model_FL::cont;p_Model_FR::cont;pTgtHydFA::cont;pTgtHydRA::cont)  {

// Detection of brake request with dependency of BLS, pMc or pTgtVeh for product ESP
// Because of either the driver or the brake system with active pressure build up have a brake demand. 

BrakeRqst_Driver = (  (pMc_available && pMc > PMc_BrakeRequest) ||
                      (!FBLS && BLS) ||
                      (BrakeRqst_Parking)
                   );

// if pMc and BLS not usable (invalid), BrakeRqstDriver_Invalid = true
BrakeRqstDriver_Invalid = (!pMc_available && FBLS);

// Brake request evaluation for system
BrakeRqst_ABS = ( (BrakeRqst_Driver) ||
                  (pTgtVeh_Enable &&  pTgtVeh > pTgtVeh_BrakeRequest)
                );


pTgtVeh_hyd = (pTgtHydFA.max(pTgtHydRA));


// Brake request evaluation for system HYD
BrakeRqst_ABS_hyd = ( (BrakeRqst_Driver) ||
                  (pTgtVeh_Enable &&  pTgtVeh_hyd > pTgtVeh_BrakeRequest)
                );

// Detection if maximum deceleration is demanded (BrakeFully_Request)
if (pTgtVeh > pTgtVeh_F)
{
	pTgtVeh_F = pTgtVeh_F + b0 * (pTgtVeh - pTgtVeh_F);
}
else
{
	pTgtVeh_F = pTgtVeh;
}

BrakeFully_Request = (pMc_available && pMc.max(pTgtVeh_F) > ( p_Model_FL.max(p_Model_FR) + dP_BrakeFully )  ) ;

}
