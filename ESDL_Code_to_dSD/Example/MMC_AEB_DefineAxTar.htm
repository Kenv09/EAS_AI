calc (ACCAEBDecCtrlAct::log;ACCAEBaxTar::cont;TrailerConnected::log;VxAct::cont;FCM_AUTO_BRK_PAR_RQ::log;FCM_BRK_ASSIST_RQ::log;AEBBAActive::log;AEBIBActive::log)  {
/* Define target deceleration AEB_axTar from DAS signals: */
/* Priority order is:                                     */
/* 1. Customer specific deceleration request AEBaxTarCus  */

/* Customer specific decleration request: */
if ((ACCAEBDecCtrlAct) || (AEBBAActive) || (AEBIBActive))
{
	AEB_axTar = ACCAEBaxTar;
}
else
{
	AEB_axTar = 0.0;	
}
/* If a trailer is connected, the deceleration must be limited, because */
/* too high decelerations could destabilize the trailer. */
if (TrailerConnected) // No trailer in MMC project then disable by setting FALSE
{
	AEB_axTar = AEB_axTar.max(C_AEB_axTarMinWithTrailer);
}

if( AEBBAActive &&(VxAct >= C_AEB_VxThreshold)) // BA & vehicle speed >=100 kph
{
	AEBFullBrake= false;
	AebPartialBrake = false;
	AEB_axTar = AEB_axTar.max(C_AEB_axTarMinThreshold);	
}
/* AEB IB handling */
else if (AEBIBActive)
{
	if(((ACCAEBDecCtrlAct == true)&&(FCM_AUTO_BRK_PAR_RQ == false)&& (FCM_BRK_ASSIST_RQ == false) && (VxAct <= C_AEB_VxPartialBrake))|| //Case: Trigger AEB IB FULL or from BA to IB FULL
		(AEBFullBrake== true))// Case:AEB IB FULL is activating.
	{
		AEBFullBrake= true;
		AebPartialBrake = false;
		AEB_axTar = AEB_axTar.max(C_AEB_axTarMin);		
	}
	else /* Otherwise: Trigger AEB IB Partial or from BA to IB Partial  */
	{
		AEBFullBrake= false;
		AebPartialBrake = true;
		AEB_axTar = AEB_axTar.max(C_AEB_axTarMinWithPartialBrake);
	}
}	
else
{
	AEBFullBrake= false;
	AebPartialBrake = false;
	AEB_axTar = AEB_axTar.max(C_AEB_axTarMin);
}
/* Limit AEB_axTar to decelerating values: */
AEB_axTar = AEB_axTar.min(0.0);

}
